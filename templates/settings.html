Пример подключения

Скопируйте этот код в ваш Python-проект (Flask):

from flask import Flask, request, jsonify
import requests

app = Flask(__name__)

WAF_URL = "(пока под вопросом)"
API_KEY = "ваш_апи_ключ"

@app.route('/', defaults={'path': ''})
@app.route('/', methods=['GET', 'POST', 'PUT', 'DELETE'])
def catch_all(path):
    print(f"Запрос к WAF: {path}")

    try:
        response = requests.request(
            method=request.method,
            url=f"{WAF_URL}/{path}",
            headers={"X-API-Key": API_KEY},
            params=request.args,
            data=request.get_data()
        )
        return response.content, response.status_code

    except requests.exceptions.ConnectionError:
        return "WAF сервис недоступен", 503

if __name__ == '__main__':
    app.run(host='127.0.0.1', port=5001, debug=True)
После запуска все запросы к вашему сайту будут проходить через WAF.
